\section{Evaluate the sum $\sum\limits_{k}\epsilon_{ijk}\epsilon_{lmk} $}

\subsection*{(a) $i = j$}

Since $\epsilon_{ijk} = 0$ when $i = j$,

\begin{equation*}
	\sum_k \epsilon_{ijk}\epsilon_{lmk} = 0
\end{equation*}

\subsection*{(b) $i = l$}

When $i = 1$, 

\begin{center}
	\begin{tabular}{cc}
	\scalebox{1.5}{
		\begin{tabular}{|c|c|c|c|}
			\hline
			\diagbox{j}{k} & 1 & 2 & 3 \\ \hline
			1 & 0 & 0 & 0 \\ \hline
			2 & 0 & 0 & 1 \\ \hline
			3 & 0 & -1& 0 \\ 
			\hline
		\end{tabular}	
	}
	&
	\scalebox{1.5}{
		\begin{tabular}{|c|c|c|c|}
			\hline
			\diagbox{m}{k} & 1 & 2 & 3 \\ \hline
			1 & 0 & 0 & 0 \\ \hline
			2 & 0 & 0 & 1 \\ \hline
			3 & 0 & -1& 0 \\ 
			\hline
		\end{tabular}	
	}
	\end{tabular}
	\vspace*{1 cm}
\end{center}

Thus, 

\begin{equation*}
	\sum_k \epsilon_{1jk}\epsilon_{1mk} = 
	\begin{cases}
	  	\ 1 & \text{if } j = m \\
	    \ 0 & \text{if } j \neq m
	  \end{cases}
\end{equation*}
\\

Likewise, when $i = 2$,

\begin{center}
	\begin{tabular}{cc}
	\scalebox{1.5}{
		\begin{tabular}{|c|c|c|c|}
			\hline
			\diagbox{j}{k} & 1 & 2 & 3 \\ \hline
			1 & 0 & 0 &-1 \\ \hline
			2 & 0 & 0 & 0 \\ \hline
			3 & 1 & 0 & 0 \\ 
			\hline
		\end{tabular}	
	}
	&
	\scalebox{1.5}{
		\begin{tabular}{|c|c|c|c|}
			\hline
			\diagbox{m}{k} & 1 & 2 & 3 \\ \hline
			1 & 0 & 0 &-1 \\ \hline
			2 & 0 & 0 & 0 \\ \hline
			3 & 1 & 0 & 0 \\ 
			\hline
		\end{tabular}	
	}
	\end{tabular}
	\vspace*{1 cm}
\end{center}

Thus, 

\begin{equation*}
	\sum_k \epsilon_{2jk}\epsilon_{2mk} = 
	\begin{cases}
	  	\ 1 & \text{if } j = m \\
	    \ 0 & \text{if } j \neq m
	  \end{cases}
\end{equation*}
\\

Last 
\begin{center}
	\begin{tabular}{cc}
	\scalebox{1.5}{
		\begin{tabular}{|c|c|c|c|}
			\hline
			\diagbox{j}{k} & 1 & 2 & 3 \\ \hline
			1 & 0 & 1 & 0 \\ \hline
			2 & -1& 0 & 0 \\ \hline
			3 & 0 & 0 & 0 \\ 
			\hline
		\end{tabular}	
	}
	&
	\scalebox{1.5}{
		\begin{tabular}{|c|c|c|c|}
			\hline
			\diagbox{m}{k} & 1 & 2 & 3 \\ \hline
			1 & 0 & 1 & 0 \\ \hline
			2 & -1& 0 & 0 \\ \hline
			3 & 0 & 0 & 0 \\ 
			\hline
		\end{tabular}	
	}
	\end{tabular}
	\vspace*{1 cm}
\end{center}

Thus, 

\begin{equation*}
	\sum_k \epsilon_{3jk}\epsilon_{3mk} = 
	\begin{cases}
	  	\ 1 & \text{if } j = m \\
	    \ 0 & \text{if } j \neq m
	  \end{cases}
\end{equation*}
\\

In sum,

\begin{equation*}
	\sum_k \epsilon_{ijk}\epsilon_{lmk} = \sum_k \epsilon_{ijk}\epsilon_{imk} = \delta_{jm} =
	\begin{cases}
	  	\ 1 & \text{if } j = m \\
	    \ 0 & \text{if } j \neq m
	  \end{cases}
\end{equation*}
